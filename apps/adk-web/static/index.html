Composition

Operator Algebra

Eight namespaces for composing every aspect of agent configuration

>>

S :: State

pick, rename, merge, flatten, validate, branch

+

C :: Context

window, summarize, relevant, extract, priority

|

T :: Tools

simple, typed, google_search, code_execution, mcp

+

P :: Prompt

role, task, constraint, format, example, persona

|

M :: Middleware

cache, dedup, metrics, before_agent, after_model

+

G :: Guards

rate_limit, toxicity, grounded, hallucination

|

E :: Evaluation

suite, response_match, trajectory, semantic

+

A :: Artifacts

publish, save, load, json_output, text_input

agent_composition.rs 
// Compose a research pipeline with operator algebra
let pipeline = AgentBuilder::new("researcher")
    .instruction("Research the given topic thoroughly")
    .google_search()
    >> AgentBuilder::new("analyst")
        .instruction("Analyze research findings")
        .temperature(0.3)
    >> AgentBuilder::new("writer")
        .instruction("Write a clear, structured report");

// Compile and run
let agent = pipeline.compile(llm);
let result = agent.run(&state).await?;
